46  void AudioSource::Sync() {
47 
48  // Set values that do not change depending on if the source is global or not:
49  alSourcef(m_Source, AL_PITCH, m_Parameters.m_Pitch);
50  alSourcef(m_Source, AL_GAIN, m_Parameters.m_GainModifier);
51  alSourcef(m_Source, AL_MIN_GAIN, m_Parameters.m_MinGain);
52  alSourcef(m_Source, AL_MAX_GAIN, m_Parameters.m_MaxGain);
53  alSourcef(m_Source, AL_ROLLOFF_FACTOR, m_Parameters.m_Rolloff);
54 
55  // If the audio source is global...
56  if (m_Parameters.m_IsGlobal) {
57 
58  /* NULLIFY EFFECT OF ORIENTATION */
59 
60  /*
61  * If AL_SOURCE_RELATIVE is true, the audio is played relative to the listener.
62  *
63  * In this case, the position of the source is (0.0f, 0.0f, 0.0f) units
64  * from the listener's position (or directly on top of the listener).
65  *
66  * This will cause the panning of the audio to always be centered, regardless of
67  * the position or orientation of the audio listener.
68  */
69  alSourcei(m_Source, AL_SOURCE_RELATIVE, AL_TRUE);
70 
71  /* SET PANNING */
72 
73  // Global AudioSources have relative position to the listener, so update local position only.
74  alSourcefv(m_Source, AL_POSITION, static_cast<ALfloat*>(&m_Parameters.m_Panning[0]));
75 
76  /* DISABLE ATTENUATION */
77 
78  // Setting reference distance to 0.0 disables attenuation.
79  alSourcef(m_Source, AL_REFERENCE_DISTANCE, 0.0f);
80 
81  // Set max distance to infinity (not strictly-necessary).
82  alSourcef(m_Source, AL_MAX_DISTANCE, std::numeric_limits<float>::max());
83 
84  // Nullify directionality of AudioSource
85  alSourcef(m_Source, AL_CONE_OUTER_ANGLE, 360.0f);
86  alSourcef(m_Source, AL_CONE_OUTER_ANGLE, 360.0f);
87 
88  /* RESET VELOCITY */
89  alListener3f(AL_VELOCITY, 0.0f, 0.0f, 0.0f);
90  }
91  else {
92 
93  // Set all defined parameters in a positional context:
94  alSourcef(m_Source, AL_REFERENCE_DISTANCE, m_Parameters.m_MinDistance);
95  alSourcef(m_Source, AL_MAX_DISTANCE, m_Parameters.m_MaxDistance);
96 
97  // Set min and max cone angles of the source.
98  alSourcef(m_Source, AL_CONE_OUTER_ANGLE, m_Parameters.m_MinAngle);
99  alSourcef(m_Source, AL_CONE_OUTER_ANGLE, m_Parameters.m_MaxAngle);
100 
101  if (const auto p = Parent().lock()) {
102  if (const auto t = p->GetComponent<Transform>().lock()) {
103 
104  // Set world position to that of the current transform.
105  alSourcefv(m_Source, AL_POSITION, static_cast<ALfloat*>(&t->m_Position[0]));
106 
107  // Set the direction of the audio source using the current transform.
108  {
109  const auto dir = t->FORWARD;
110 
111  alSource3f(m_Source, AL_DIRECTION, dir.x, dir.y, dir.z);
112  }
113 
114  // Set velocity (using rigidbody, if available):
115  {
116  glm::vec3 velocity;
117 
118  if (const auto r = p->GetComponent<Physics::Rigidbody>().lock()) {
119  velocity = r->Velocity();
120  }
121  else {
122  velocity = (t->m_Position - m_LastPosition) * Time::DeltaTime();
123  }
124 
125  alListenerfv(AL_VELOCITY, static_cast<ALfloat*>(&velocity[0]));
126  }
127 
128  m_LastPosition = t->m_Position;
129  }}
130  }

175  void AudioSource::Play(const bool& _allowFallback) {
176 
177  try {
178 
179  // Try and get current clip.
180  if (const auto c = m_Clip.lock()) {
181 
182  // Only play if the clip actually contains data.
183  if (c->m_Samples.m_Length > 0) {
184 
185  if (c->m_ALBuffer == AL_NONE) {
186 
187  /*
188  * AL buffer non-existent. Implies AL failed to initialise correctly.
189  * Attempt to play the clip globally instead. This will use whatever fallback is available.
190  */
191 
192  if (_allowFallback) {
193  Sound::PlayGlobal(c);
194  }
195  else {
196  throw std::runtime_error(
197  "Cannot play audio clip since m_ALBuffer == AL_NONE, "
198  "and permission to use a fallback is denied!\n"
199  );
200  }
201  }
202  else {
203 
204  // If already playing, nothing needs to be done.
205  if (State() != AL_PLAYING) {
206 
207  // Play!
208  alSourcei(m_Source, AL_BUFFER, static_cast<ALint>(c->m_ALBuffer));
209  alSourcePlay(m_Source);
210  }
211  else {
212  throw std::runtime_error("Attempted to call Play() on an AudioSource that is already playing.");
213  }
214  }
215  }
216  else {
217  throw std::runtime_error("Cannot play AudioClip since its size is zero!");
218  }
219  }
220  else {
221  throw std::runtime_error("Cannot play AudioSource since the current clip is nullptr!");
222  }
223  }
224  catch (const std::exception& e) {
225  std::cout << e.what() << '\n';
226  }
