44  void AudioSource::Sync() {
45 
46  // Set values that do not change depending on if the source is global or not:
47  alSourcef(m_Source, AL_PITCH, m_Parameters.m_Pitch);
48  alSourcef(m_Source, AL_GAIN, m_Parameters.m_GainModifier);
49  alSourcef(m_Source, AL_MIN_GAIN, m_Parameters.m_MinGain);
50  alSourcef(m_Source, AL_MAX_GAIN, m_Parameters.m_MaxGain);
51  alSourcef(m_Source, AL_ROLLOFF_FACTOR, m_Parameters.m_Rolloff);
52 
53  // If the audio source is global...
54  if (m_Parameters.m_IsGlobal) {
55 
56  /* NULLIFY EFFECT OF ORIENTATION */
57 
58  /*
59  * If AL_SOURCE_RELATIVE is true, the audio is played relative to the listener.
60  *
61  * In this case, the position of the source is (0.0f, 0.0f, 0.0f) units
62  * from the listener's position (or directly on top of the listener).
63  *
64  * This will cause the panning of the audio to always be centered, regardless of
65  * the position or orientation of the audio listener.
66  */
67  alSourcei(m_Source, AL_SOURCE_RELATIVE, AL_TRUE);
68 
69  /* SET PANNING */
70 
71  // Global AudioSources have relative position to the listener, so update local position only.
72  alSourcefv(m_Source, AL_POSITION, static_cast<ALfloat*>(&m_Parameters.m_Panning[0]));
73 
74  /* DISABLE ATTENUATION */
75 
76  // Setting reference distance to 0.0 disables attenuation.
77  alSourcef(m_Source, AL_REFERENCE_DISTANCE, 0.0f);
78 
79  // Set max distance to infinity (not strictly-necessary).
80  alSourcef(m_Source, AL_MAX_DISTANCE, std::numeric_limits<float>::max());
81 
82  // Nullify directionality of AudioSource
83  alSourcef(m_Source, AL_CONE_OUTER_ANGLE, 360.0f);
84  alSourcef(m_Source, AL_CONE_OUTER_ANGLE, 360.0f);
85 
86  /* RESET VELOCITY */
87  alListener3f(AL_VELOCITY, 0.0f, 0.0f, 0.0f);
88  }
89  else {
90 
91  // Set all defined parameters in a positional context:
92  alSourcef(m_Source, AL_REFERENCE_DISTANCE, m_Parameters.m_MinDistance);
93  alSourcef(m_Source, AL_MAX_DISTANCE, m_Parameters.m_MaxDistance);
94 
95  // Set min and max cone angles of the source.
96  alSourcef(m_Source, AL_CONE_OUTER_ANGLE, m_Parameters.m_MinAngle);
97  alSourcef(m_Source, AL_CONE_OUTER_ANGLE, m_Parameters.m_MaxAngle);
98 
99  // Get reference to parent (may be null).
100  const auto parent = Parent();
101 
102  if (parent != nullptr) {
103 
104  // Get reference to transform (may be null).
105  const auto transform = parent->GetComponent<Transform>();
106 
107  if (transform != nullptr) {
108 
109  // Set world position to that of the current transform.
110  alSourcefv(m_Source, AL_POSITION, static_cast<ALfloat*>(&transform->m_Position[0]));
111 
112  // Set the direction of the audio source using the current transform.
113  {
114  const auto dir = transform->FORWARD;
115 
116  alSource3f(m_Source, AL_DIRECTION, dir.x, dir.y, dir.z);
117  }
118  }
119 
120  // Set velocity (using rigidbody, if available):
121  {
122  glm::vec3 velocity;
123 
124  const auto rigidbody = parent->GetComponent<Rigidbody>();
125 
126  if (rigidbody != nullptr) {
127  velocity = rigidbody->Velocity();
128  }
129  else {
130  velocity = (transform->m_Position - m_LastPosition) * Time::DeltaTime();
131  }
132 
133  alListenerfv(AL_VELOCITY, static_cast<ALfloat*>(&velocity[0]));
134  }
135 
136  m_LastPosition = transform->m_Position;
137  }
138  }

183  void AudioSource::Play(const bool& _allowFallback) {
184 
185  try {
186 
187  // Try and get current clip.
188  const auto c = m_Clip.lock();
189 
190  if (c != nullptr) {
191 
192  // Only play if the clip actually contains data.
193  if (c->m_Samples.m_Length > 0) {
194 
195  if (c->m_ALBuffer == AL_NONE) {
196 
197  /*
198  * AL buffer non-existent. Implies AL failed to initialise correctly.
199  * Attempt to play the clip globally instead. This will use whatever fallback is available.
200  */
201 
202  if (_allowFallback) {
203  Sound::PlayGlobal(c);
204  }
205  else {
206  throw std::runtime_error(
207  "Cannot play audio clip since m_ALBuffer == AL_NONE, "
208  "and permission to use a fallback is denied!\n"
209  );
210  }
211  }
212  else {
213 
214  // If already playing, nothing needs to be done.
215  if (State() != AL_PLAYING) {
216 
217  // Play!
218  alSourcei(m_Source, AL_BUFFER, static_cast<ALint>(c->m_ALBuffer));
219  alSourcePlay(m_Source);
220  }
221  else {
222  throw std::runtime_error("Attempted to call Play() on an AudioSource that is already playing.");
223  }
224  }
225  }
226  else {
227  throw std::runtime_error("Cannot play AudioClip since its size is zero!");
228  }
229  }
230  else {
231  throw std::runtime_error("Cannot play AudioSource since the current clip is nullptr!");
232  }
233  }
234  catch (const std::exception& e) {
235  std::cout << e.what() << '\n';
