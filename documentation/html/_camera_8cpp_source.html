1000  void Camera::Blur(const RenderTexture& _rt, const float& _intensity, const int& _passes, const bool& _highQuality, const bool& _consistentDPI) {
1001 
1002  // Get dimensions of screen.
1003  const auto dimensions = glm::vec2(_rt.Width(), _rt.Height());
1004 
1005  // Horizontal and vertical blur passes.
1006  if (const auto h = Resources::GetShader("blur_horizontal").lock()) {
1007  if (const auto v = Resources::GetShader("blur_vertical" ).lock()) {
1008 
1009  // Assign target texture to both shader programs:
1010  h->Assign(h->AttributeID("u_Texture"), _rt.ID(), 0, GL_TEXTURE_2D);
1011  v->Assign(v->AttributeID("u_Texture"), _rt.ID(), 0, GL_TEXTURE_2D);
1012 
1013  // Optionally, use a multiplier proportional to the screen's resolution to
1014  // keep the blur effect consistent across a range of different resolutions.
1015  const float dpiFactor = _consistentDPI ?
1016  glm::sqrt(dimensions.x * dimensions.y) * (1.0f / 3000.0f) : // (1.0f / 3000.0f) is an arbitrary value I chose by eye.
1017  1.0f;
1018 
1019  float rootIntensity, scalar, aspectCorrection;
1020 
1021  // Init rootIntensity, scalar, and aspectCorrection.
1022  if (_highQuality) {
1023  rootIntensity = 0.0f;
1024  scalar = 0.0f;
1025  aspectCorrection = 0.0f;
1026  }
1027  else {
1028  rootIntensity = glm::sqrt(_intensity);
1029  scalar = glm::sqrt(rootIntensity);
1030  aspectCorrection = glm::sqrt(dimensions.x / (float)dimensions.y);
1031  }
1032 
1033  // Perform a blur pass:
1034  for (int i = 0; i < _passes; i++) {
1035 
1036  int width, height;
1037 
1038  float size;
1039 
1040  // Init width, height and size.
1041  if (_highQuality) {
1042  width = (int)dimensions.x;
1043  height = (int)dimensions.y;
1044 
1045  size = (float)glm::pow(1.618f, i + 1) * _intensity;
1046  }
1047  else {
1048  width = (int)glm::ceil(dimensions.x / glm::pow(2.0f, (float)i / 2.0f * scalar));
1049  height = (int)glm::ceil(dimensions.y / glm::pow(2.0f, (float)i / 2.0f * scalar) * aspectCorrection);
1050 
1051  size = (float)i * rootIntensity;
1052  }
1053 
1054  // Assign step value to both shader programs.
1055  h->Assign(h->AttributeID("u_Step"), dpiFactor * size);
1056  v->Assign(v->AttributeID("u_Step"), dpiFactor * size);
1057 
1058  // Blit into a temporary texture, and blur that once on the x, and once on the y.
1059  const RenderTexture tmp(width, height, _rt.Format(), Texture::Parameters::FilterMode(GL_LINEAR, GL_LINEAR), _rt.WrapMode(), RenderTexture::Parameters::DepthMode::NONE);
1060  Blit(_rt, tmp, h);
1061  Blit(tmp, tmp, v);
1062 
1063  // Copy the temp texture back to the original target.
1064  Copy(tmp, _rt);
1065 
1066  // Break early if the width or height are 1, as additional blurring is practically pointless.
1067  if (width == 1 && height == 1) {
1068  break;
1069  }
1070  }
1071  }}

1089  if (const auto mask = Resources::GetTexture("exposure_weights").lock()) {
1090

1092  Shader::Bind(s->ID());
1093  s->Assign(s->AttributeID("u_Weights"), mask->ID(), 1, GL_TEXTURE_2D);
1094  }
1095 
1096  Blit(m_RT, luma_out, s);
1097

1099  std::vector<float> pixels(static_cast<size_t>(luma_res.x * luma_res.y * luma_out.Format().Channels()));
1100 
1101  // Load the luminosity samples into the buffer.
1102  RenderTexture::Bind(luma_out);
1103  glReadPixels(0, 0, luma_res.x, luma_res.y, luma_out.Format().TextureFormat(), GL_FLOAT, pixels.data());
1104

1106  int avg_count = 0;
1107  auto avg = 0.0f;
1108 
1109  for (auto y = 0; y < luma_res.y; y++) {
1110  for (auto x = 0; x < luma_res.x; x++) {
1111 
1112  const auto l = pixels.at((y * luma_res.x) + x);
1113 
1114  if (l > 0) {
1115  avg += l;
1116 
1117  avg_count++;
1118  }
1119  }}
1120 
1121  // Detect and prevent NaN propagation by checking the value against itself:
1122  if (avg != avg) {
1123  avg = 0.0f;
1124  }
1125 
1126  avg = avg / (float)glm::max(avg_count, 1);
1127 
1128  // Get difference between current exposure and average luma.
1129  const float diff = glm::clamp(
1130  (Settings::PostProcessing::ToneMapping::s_Exposure + target::s_Compensation) - avg,
1131  -1.0f,
1132  1.0f
1133  );

1137  target::s_SpeedUp :
1138  target::s_SpeedDown;
1139

1141  m_Exposure = glm::mix(
1142  m_Exposure,
1143  glm::clamp(Settings::PostProcessing::ToneMapping::s_Exposure + diff, target::s_MinEV, target::s_MaxEV),
1144  Time::DeltaTime() * speed
1145  );
1146  }
