859  void Camera::Blur(const RenderTexture& _rt, const float& _intensity, const int& _passes, const bool& _highQuality, const bool& _consistentDPI) {
860 
861  // Get dimensions of screen.
862  const auto dimensions = glm::vec2(_rt.Width(), _rt.Height());
863 
864  // Horizontal and vertical blur passes.
865  auto horizontal = Resources::GetShader("blur_horizontal");
866  auto vertical = Resources::GetShader("blur_vertical");
867 
868  // Assign target texture to shader programs:
869  horizontal.lock()->Assign(horizontal.lock()->AttributeID("u_Texture"), _rt.ID());
870  vertical.lock()->Assign( vertical.lock()->AttributeID("u_Texture"), _rt.ID());
871 
872  // Screen width and height as integers.
873  const int w = (int)dimensions.x,
874  h = (int)dimensions.y;
875 
876  // Optionally, use a multiplier proportional to the screen's resolution to
877  // keep the blur effect consistent across a range of different resolutions.
878  const float dpiFactor = _consistentDPI ?
879  glm::sqrt(dimensions.x * dimensions.y) * (1.0f / 3000.0f) : // (1.0f / 3000.0f) is an arbitrary value I chose by eye.
880  1.0f;
881 
882  float rootIntensity, scalar, aspectCorrection;
883 
884  // Init rootIntensity, scalar, and aspectCorrection.
885  if (_highQuality) {
886  rootIntensity = 0.0f;
887  scalar = 0.0f;
888  aspectCorrection = 0.0f;
889  }
890  else {
891  rootIntensity = glm::sqrt(_intensity);
892  scalar = glm::sqrt(rootIntensity);
893  aspectCorrection = glm::sqrt(dimensions.x / (float)dimensions.y);
894  }
895 
896  // Perform a blur pass:
897  for (int i = 0; i < _passes; i++) {
898 
899  int width, height;
900 
901  float size;
902 
903  // Init width, height and size.
904  if (_highQuality) {
905  width = w;
906  height = h;
907 
908  size = (float)glm::pow(1.618f, i + 1) * _intensity;
909  }
910  else {
911  width = (int)glm::ceil(float(w) / glm::pow(2.0f, (float)i / 2.0f * scalar));
912  height = (int)glm::ceil(float(h) / glm::pow(2.0f, (float)i / 2.0f * scalar) * aspectCorrection);
913 
914  size = (float)i * rootIntensity;
915  }
916 
917  // Assign step value to shader programs.
918  horizontal.lock()->Assign(horizontal.lock()->AttributeID("u_Step"), dpiFactor * size);
919  vertical.lock()->Assign( vertical.lock()->AttributeID("u_Step"), dpiFactor * size);
920 
921  // Blit into a temporary texture, and blur that once on the x, and once on the y.
922  RenderTexture tmp(width, height, _rt.Format(), Texture::Parameters::FilterMode(GL_LINEAR, GL_LINEAR), _rt.WrapMode(), RenderTexture::Parameters::DepthMode::NONE);
923  Blit(_rt, tmp, horizontal);
924  Blit(tmp, tmp, vertical);
925 
926  // Copy the temp texture back to the original target.
927  Copy(tmp, _rt);
928 
929  // Break early if the width or height are 1, as additional blurring is practically pointless.
930  if (width == 1 && height == 1) {
931  break;
932  }
933  }
934  }

936  void Camera::AutoExposure() {
937 
938  using target = Settings::PostProcessing::ToneMapping::AutoExposure;
939 
940  // Get window dimensions:
941  const auto dimensions = GetWindow()->Dimensions();
942 
943  // Load shader program:
944  auto shader = Resources::GetShader("auto_exposure");
945 
946  // Create a 32 by 32 render texture for the luminosity calculations.
947  const glm::ivec2 luma_res(32, 32);
948 
949  RenderTexture luma_out(luma_res.x, luma_res.y, Texture::Parameters::Format(m_RT.Format().PixelFormat(), false), Texture::Parameters::FilterMode(GL_LINEAR, GL_NEAREST), m_RT.WrapMode(), RenderTexture::Parameters::DepthMode::NONE);
950 
951  // Load a mask for the average luminosity calculation.
952  auto mask = Resources::GetTexture("exposure_weights");
953 
954  // Generate the luminosity texture:
955  Shader::Bind(shader.lock()->ID());
956  shader.lock()->Assign(shader.lock()->AttributeID("u_Weights"), mask.lock()->ID(), 1, GL_TEXTURE_2D);
957 
958  Blit(m_RT, luma_out, shader);
959 
960  // Create a buffer for the luminosity samples:
961  std::vector<float> pixels(static_cast<size_t>(luma_res.x * luma_res.y * luma_out.Format().Channels()));
962 
963  // Load the luminosity samples into the buffer.
964  RenderTexture::Bind(luma_out);
965  glReadPixels(0, 0, luma_res.x, luma_res.y, luma_out.Format().TextureFormat(), GL_FLOAT, pixels.data());
966 
967  // Compute the average luminosity:
968  int avg_count = 0;
969  float avg = 0.0f;
970 
971  for (auto y = 0; y < luma_res.y; y++) {
972  for (auto x = 0; x < luma_res.x; x++) {
973 
974  float l = pixels.at((y * luma_res.x) + x);
975 
976  if (l > 0) {
977  avg += l;
978 
979  avg_count++;
980  }
981  }}
982 
983  // Detect and prevent NaN propagation by checking the value against itself:
984  if (avg != avg) {
985  avg = 0.0f;
986  }
987 
988  avg = avg / (float)glm::max(avg_count, 1);
989 
990  // Get difference between current exposure and average luma.
991  const float diff = glm::clamp(
992  (Settings::PostProcessing::ToneMapping::s_Exposure + target::s_Compensation) - avg,
993  -1.0f,
994  1.0f
995  );
996 
997  // Determine the speed to change exposure by:
998  const float speed = (diff - m_Exposure) >= 0 ?
999  target::s_SpeedUp :
1000  target::s_SpeedDown;
1001 
1002  // Set a new exposure value:
1003  m_Exposure = glm::mix(
1004  m_Exposure,
1005  glm::clamp(Settings::PostProcessing::ToneMapping::s_Exposure + diff, target::s_MinEV, target::s_MaxEV),
1006  Time::DeltaTime() * speed
1007  );
1008  }
