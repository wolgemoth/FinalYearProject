148  {

151

156 
157  GLenum shaderType;
158 
159  if (strcmp(item.extension().string().c_str(), ".vert") == 0) { shaderType = GL_VERTEX_SHADER; }
160  else if (strcmp(item.extension().string().c_str(), ".frag") == 0) { shaderType = GL_FRAGMENT_SHADER; }
161  else if (strcmp(item.extension().string().c_str(), ".geom") == 0) { shaderType = GL_GEOMETRY_SHADER; }
162  else if (strcmp(item.extension().string().c_str(), ".glsl") == 0) { shaderType = GL_SHADER; }
163  else {
164  shaderType = GL_NONE;
165  }
166

177  separated.Assign(item.stem().string(), subshaders);
178  }
179  }
