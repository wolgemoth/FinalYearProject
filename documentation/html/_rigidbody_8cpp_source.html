94  float referenceArea = 4.0f * (glm::pi<float>() * (_radius * _radius)) / 2.0f;
95 
96  // Courtesy of: https://www.grc.nasa.gov/WWW/k-12/airplane/drageq.html
97  // This code snippet initializes a 3D vector called "result" with all components set to 0.0. It then calculates the magnitude of a velocity vector and stores it in the variable "len".
98  float v = glm::length2(Velocity());

108  glm::vec3 Rigidbody::FrictionForce(Collision& _collision) {
109 
110  glm::vec3 result(0.0f);
111 
112  const float coefficient = 0.2f;
113 
114  // Velocity vector and its magnitude.
115  auto dir = Velocity();
116  auto len = glm::length(dir);
117 
118  if (len != 0) {
119 
120  // Project the velocity onto the normal.
121  auto proj = dir - _collision.Normal() * glm::dot(dir, _collision.Normal());
122 
123  // Objects with more momentum going "into" a surface should receive more friction.
124  float response = 1.0f - (glm::dot(proj, dir) / len);
125 
126  // Friction should never speed up an object.

128
