160  void Rigidbody::Interpolate() {
161 
162  if (Physics::s_LastTick > 0.0f) {
163 
164  const auto transform = m_Transform.lock();
165 
166  if (transform != nullptr) {
167 
168  // Get the transform from the bullet physics engine.
169  auto t = m_Parameters.m_BulletRigidbody->m_Rigidbody->getWorldTransform();
170 
171  // Get the position and rotation of the rigidbody as they exist within bullet.
172  const auto bOrigin = t.getOrigin();
173  const auto bRotation = t.getRotation().inverse();
174 
175  // Convert to glm types.
176  const auto lastPos = glm::vec3( bOrigin.x(), bOrigin.y(), bOrigin.z());
177  const auto lastRot = glm::quat(bRotation.w(), bRotation.x(), bRotation.y(), bRotation.z());
178 
179  // Compute quaternion from angular velocity.
180  const auto av = AngularVelocity();
181 
182  glm::quat rot;
183  if (glm::length(av) > 0.005f) {
184  rot = glm::angleAxis(glm::cos(glm::length(av)) * Time::FixedDeltaTime(), glm::normalize(av));
185  }
186  else {
187  rot = glm::identity<glm::quat>();
188  }
189 
190  // Apply new values. Linearly interpolate using length of time since last physics update.
191  transform->m_Position = lastPos + (Velocity() * Physics::s_LastTick * Time::FixedDeltaTime());
192  transform->m_Rotation = glm::slerp(lastRot, lastRot * rot, glm::max(Physics::s_LastTick, 0.0f));
193  }
194  }

220  void Rigidbody::Sync() {
221 
222  try {
223 
224  /* SYNCHRONISE POSITION AND ORIENTATION */
225  {
226  const auto transform = m_Transform.lock();
227 
228  if (transform != nullptr) {
229 
230  // Get the transform from the bullet physics engine.
231  auto t = m_Parameters.m_BulletRigidbody->m_Rigidbody->getWorldTransform();
232 
233  const auto bOrigin = t.getOrigin();
234  const auto bRotation = t.getRotation().inverse();
235 
236  // Sync the transform from bullet with the transform in-engine.
237  transform->m_Position = glm::vec3( bOrigin.x(), bOrigin.y(), bOrigin.z());
238  transform->m_Rotation = glm::quat(bRotation.w(), bRotation.x(), bRotation.y(), bRotation.z());
239  }
240  }
241 
242  /* GET COLLISIONS */
243  {
244  // Reset the collisions.
245  m_Collisions.clear();
246 
247  /*
248  * TODO: Move this to the Physics class and have some kind of managed, internal
249  * system for indexing Rigidbody objects against btRigidbody objects.
250  */
251 
252  // Get and iterate through every contact manifold in the physics engine.
253  int numManifolds = Physics::s_DynamicsWorld->getDispatcher()->getNumManifolds();
254  for (int i = 0; i < numManifolds; i++) {
255 
256  auto* contactManifold = Physics::s_DynamicsWorld->getDispatcher()->getManifoldByIndexInternal(i);
257 
258  // Get and iterate through every contact in the manifold.
259  int numContacts = contactManifold->getNumContacts();
260  for (int j = 0; j < numContacts; j++) {
261 
262  // Get a pointer to the btRigidbody associated with this rigidbody.
263  const auto* rbThis = m_Parameters.m_BulletRigidbody->m_Rigidbody.get();
264 
265  // Validate the pointer. If is fails the null check we will throw.
266  if (rbThis != nullptr) {
267 
268  // Determine if either of the bodies in the manifold are this rigidbody.
269  int body;
270 
271  if (contactManifold->getBody0() == rbThis) { body = 0; }
272  else if (contactManifold->getBody1() == rbThis) { body = 1; }
273  else { body = -1; }
274 
275  // If one of the rigidbodies involved in the collision event are this rigidbody,
276  // generate a collision object and append it to the list of collisions.
277  if (body >= 0) {
278  m_Collisions.emplace_back(contactManifold->getContactPoint(j), body);
279  }
280  }
281  else {
282  throw std::runtime_error("btRigidbody associated with this Rigidbody is nullptr!");
283  }
284  }
285  }
286  }
287  }
288  catch (const std::exception& e) {
289  std::cout << e.what() << '\n';
